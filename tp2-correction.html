def best_knn_score(X,y):
    k_range=range(1,21)
    score_test = []
    X, y = shuffle(X,y)
    X_train = X[:90]
    y_train = y[:90]
    X_test = X[90:]
    y_test = y[90:]
    for k in k_range:
       knn =  KNeighborsClassifier(n_neighbors=k)
       knn.fit(X_train,y_train)
       score_test.append(knn.score(X_test,y_test))
    return max(score_test)

print(best_knn_score(X,y))
print(best_knn_score(X,y))
print(best_knn_score(X,y))

0.9833333333333333
0.95
0.9333333333333333

Le rsultat est diffrent chaque fois. Cela est d au mlange alatoire des donnes par shuffle(X,y).
